$(document).ready(function () {
    var option = {
        onKeyPress: function (tel, e, field, options) {
            var masks = ['(00) 0000 0000', '(00) 00000 0000'];
            mask = (tel.length > 14) ? masks[1] : masks[0];
            $('[name=Sobrenome]').mask(mask, options);
        }
    };
    $('[name=Sobrenome]').mask('(00) 0000 0000', option);
});

$("#Form").validate({
    rules: {
        Telefone: {
            required: true,
			minlength: 14
        }
    },
    messages: {
        Telefone: {
            minlength: "Inválido. Tente no formato de: “(12) 1234 - 1234”"
        }
    }
});
